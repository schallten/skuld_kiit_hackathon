// Generate content using OpenAI with automatic retries
export const generateContent = task({
  id: "generate-content",
  retry: {
    maxAttempts: 3,
    exponentialBackoff: true,
  },
  queue: { concurrency: 10 },
  run: async ({ theme, description, userId }) => {
    // Generate text content
    const textResult = await openai.chat.completions.create({
      model: "gpt-4o",
      messages: generateTextPrompt(theme, description),
      stream: true, // Stream responses to users
    });

    // Generate accompanying image
    const imageResult = await openai.images.generate({
      model: "dall-e-3",
      prompt: generateImagePrompt(theme, description),
      quality: "hd",
    });

    // Send real-time updates to frontend
    await io.emit(`content-${userId}`, {
      text: textResult.choices[0].message.content,
      image: imageResult.data[0].url,
      status: "completed"
    });

    return { success: true, contentId: generateId() };
  },
});
